```typescript
import { GameState } from '../state/GameState';
import { SaveManager } from '../state/SaveManager';

static transitionToRoom(scene, roomId, spawnPoint, transition, duration): void {
    // Auto-save before leaving the current room
    const state = GameState.getInstance();
    // Update current room to the DESTINATION room before saving
    // This way, loading auto-save will start at the room we're heading to
    state.getData().currentRoom = roomId;
    state.markRoomVisited(roomId);
    SaveManager.autoSave(state);

    scene.input.enabled = false;
    // ... rest of existing transition code unchanged ...
}
```

IMPORTANT: Auto-save records the DESTINATION room, not the current room. This means death in the new room restores to the beginning of that room (the player just entered). This matches research Pitfall #1: "auto-save on room entry, death resets to room-entry state."

Note: Direct mutation of `state.getData().currentRoom` works because getData() returns the internal data object (the Readonly type annotation is for compile-time safety only). If the implementation returns a copy, add a `setCurrentRoom(roomId)` method to GameState instead.

```typescript
this.commandSubmittedHandler = (text: string) => {
    if (this.isTransitioning) return;

    // Get current inventory item info for noun resolution
    const inventoryItems = this.itemDefs
        .filter(item => this.gameState.hasItem(item.id))
        .map(item => ({ id: item.id, name: item.name }));

    const parseResult = this.textParser.parse(
        text,
        this.roomData.hotspots,
        this.roomData.exits,
        inventoryItems,
    );

    if (!parseResult.success || !parseResult.action) {
        this.narratorDisplay.typewrite(
            parseResult.error ?? `I don't understand "${text}". Try commands like 'look', 'take', 'go', or 'use'.`
        );
        return;
    }

    const result = this.commandDispatcher.dispatch(parseResult.action, this.roomData);

    // Use typewriter for narrator-style responses, instant for short system messages
    if (result.response.length > 50) {
        this.narratorDisplay.typewrite(result.response);
    } else {
        this.narratorDisplay.showInstant(result.response);
    }

    // Update inventory panel after each command (items may have changed)
    this.updateInventoryPanel();
};
```
